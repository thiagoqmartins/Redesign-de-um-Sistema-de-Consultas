Consulta Exemplo: Quem cuida de "CÁLCULO" em "REDUTORES"?

SELECT a.nome AS atividade, ar.nome AS area, r.nome AS responsavel, ar2.posicao
FROM atividade_responsavel ar2
JOIN atividade a ON a.id_atividade = ar2.id_atividade
JOIN area ar ON ar.id_area = ar2.id_area
JOIN responsavel r ON r.id_responsavel = ar2.id_responsavel
WHERE a.nome = 'CÁLCULO' AND ar.nome = 'REDUTORES';
